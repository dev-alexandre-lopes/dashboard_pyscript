def plot_cosseno():
        x2 = np.linspace(0, 4 * np.pi, 100)
        y2 = np.cos(x2)
        fig, ax2 = plt.subplots()
        ax2.plot(x2,y2)
        plt.title("Gráfico da Função Cosseno")
        display(fig, target="graph-area2", append=False)
  
  
  plot_cosseno()
